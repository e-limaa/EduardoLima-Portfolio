## Uso

```tsx
import { ProjectCard } from "@antigravity/ds";

<ProjectCard
  title="Nome do Projeto"
  category="Design System"
  image="/caminho/para/imagem.jpg"
  metric="2024"
  color="from-blue-500 to-cyan-500"
  onClick={() => console.log("Clicado")}
/>
```
