## Usage

```tsx
import { ProjectCard } from "@antigravity/ds";

<ProjectCard
  title="Project Name"
  category="Design System"
  image="/path/to/image.jpg"
  metric="2024"
  color="from-blue-500 to-cyan-500"
  onClick={() => console.log("Clicked")}
/>
```
